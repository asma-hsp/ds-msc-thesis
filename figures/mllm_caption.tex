%
% Usage in background.tex:
%   \begin{figure}[t]\centering
%     \resizebox{0.95\textwidth}{!}{\input{figures/mllm_caption}}
%     \caption{...}\label{fig:bg_mllm}
%   \end{figure}

\begin{tikzpicture}[
  >=Stealth, line cap=round, line join=round, font=\small,
  imgph/.style = {draw=gray!55, fill=gray!8, rounded corners=3pt,
                  minimum width=1.7cm, minimum height=1.3cm,
                  align=center, font=\scriptsize, text=gray!55},
  txtph/.style = {draw=gray!50, fill=yellow!12, rounded corners=3pt,
                  minimum width=2.5cm, minimum height=0.9cm,
                  align=center, font=\scriptsize\itshape, inner sep=3pt},
  block/.style = {draw=#1!65!black, fill=#1!10, rounded corners=3pt,
                  minimum width=2.0cm, minimum height=0.9cm, align=center},
  capph/.style = {draw=orange!55, fill=orange!8, rounded corners=3pt,
                  minimum width=3.2cm, minimum height=0.95cm, align=center,
                  font=\scriptsize\itshape, inner sep=4pt},
  arr/.style   = {->, draw=gray!55, line width=0.8pt},
]

% ---- inputs ----
\node[imgph] (img) at (0,1.1) {$3{\times}3$ tiled\\ ref.\ image};
\node[below=1pt of img, font=\scriptsize] {$q^v$};
\node[txtph] (txt) at (0,-1.1) {``in the high-top\\ version''};
\node[below=1pt of txt, font=\scriptsize] {$q^t$};

% ---- MLLM internals ----
\begin{scope}[on background layer]
  \draw[draw=orange!40, fill=orange!4, rounded corners=5pt]
    (2.2,-1.75) rectangle (7.3,1.75);
\end{scope}
\node[block=blue]   (vit)  at (3.4,1.1)  {Vision\\ encoder (ViT)};
\node[block=teal]   (conn) at (5.9,1.1)  {MLP\\ connector};
\node[block=purple] (llm)  at (5.9,-1.1) {LLM};
\node[font=\scriptsize\itshape, text=orange!70!black] at (3.4,2.35) {MLLM (InternVL-3.5-8B)};

\draw[arr] (img.east) -- (vit.west);
\draw[arr] (vit.east) -- (conn.west);
\draw[arr] (conn.south) -- (llm.north);          % visual tokens -> LLM
\draw[arr] (txt.east) -- (llm.west);             % text prompt -> LLM

% ---- output caption ----
\node[capph] (cap) at (10.0,-1.1)
  {``A high-top sneaker\\ with the same colourway''};
\node[below=1pt of cap, font=\scriptsize] {target caption $c$};
\draw[arr, draw=orange!65!black] (llm.east) -- (cap.west);

\end{tikzpicture}
