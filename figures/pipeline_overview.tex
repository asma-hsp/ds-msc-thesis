\def\pipeTop{figures/pipeline_samples/top1}     % top-1 retrieved image
\def\pipeDbA{figures/pipeline_samples/db1}      % image-dataset thumbnails
\def\pipeDbB{figures/pipeline_samples/db2}
\def\pipeDbC{figures/pipeline_samples/db3}
\def\pipeQueryText{in an old archival photo}
\def\pipeCaptionText{``Pantheon in an old archival photo with large columns and inscribed facade''}

\begin{tikzpicture}[
  >=Stealth, line cap=round, line join=round,
  font=\normalsize,
  % ── node styles ───────────────────────────────────────────────────────
  imgnode/.style  = {draw=#1!70!black, line width=1.1pt, inner sep=0pt,
                      rounded corners=1pt},
  imgnode/.default = gray,
  enc/.style      = {draw=#1!65!black, fill=#1!10, rounded corners=3pt,
                      minimum width=2.05cm, minimum height=0.72cm,
                      align=center, inner sep=3pt, font=\normalsize},
  enc/.default    = gray,
  txtph/.style    = {draw=gray!55, fill=yellow!12, rounded corners=3pt,
                      text width=2.15cm, minimum height=0.80cm,
                      align=center, font=\small\itshape, inner sep=4pt},
  capph/.style    = {draw=orange!55, fill=orange!8, rounded corners=3pt,
                      text width=2.55cm, align=center,
                      font=\footnotesize\itshape, inner sep=5pt},
  op/.style       = {draw=gray!60, fill=white, circle,
                      minimum size=0.60cm, inner sep=0pt, font=\small},
  dotop/.style    = {draw=gray!60, fill=white, circle,
                      minimum size=0.58cm, inner sep=0pt, font=\Large\bfseries},
  prodop/.style   = {draw=gray!60, fill=gray!8, circle,
                      minimum size=0.72cm, inner sep=0pt, font=\large\bfseries},
  lbl/.style      = {font=\scriptsize\itshape, text=#1},
  lbl/.default    = {gray!65!black},
  sub/.style      = {font=\small},
  % ── arrow styles ──────────────────────────────────────────────────────
  iarr/.style     = {->, draw=blue!60!black,   line width=0.95pt},
  tarr/.style     = {->, draw=teal!65!black,   line width=0.95pt},
  carr/.style     = {->, draw=orange!70!black, line width=0.95pt},
  garr/.style     = {->, draw=gray!55,         line width=0.80pt},
]

% ── branch y-coordinates ──────────────────────────────────────────────────────
\def\yI{3.0}    % image branch
\def\yT{0.0}    % text branch
\def\yC{-3.0}   % caption branch

% =============================================================================
% INPUTS  (x = 0): image query + text query
% =============================================================================
\node[imgnode=blue] (qv) at (0, \yI) {\includegraphics[width=1.75cm]{\pipeRef}};
\node[below=1pt of qv, sub] {image query $q^v$};

\node[txtph] (qt) at (0, \yT) {\pipeQueryText};
\node[below=1pt of qt, sub] {text query $q^t$};

% =============================================================================
% MLLM caption generation:  q^v + q^t  ->  MLLM  ->  caption  (flows forward)
% =============================================================================
\node[enc=orange, minimum width=2.15cm] (mllm) at (2.55, 1.45) {InternVL\\3.5-8B};
\node[above=1pt of mllm, lbl=orange!75!black] {MLLM};

% both inputs feed the MLLM (orange = caption-generation step)
\draw[carr] (qv.east) -- (mllm.north west);
\draw[carr] (qt.east) -- (mllm.south west);

% caption emerges to the right of / below the MLLM, then flows into its encoder
\node[capph] (cap) at (3.7, \yC) {\pipeCaptionText};
\node[below=1pt of cap, sub] {target caption $c$};
\draw[carr] (mllm.east) -| (cap.north);

% =============================================================================
% ENCODERS  (x = 6.1)
% =============================================================================
\def\encX{6.6}
\node[enc=blue]   (ev) at (\encX, \yI) {visual encoder\\$\phi^v$};
\node[enc=teal]   (et) at (\encX, \yT) {text encoder\\$\phi^t$};
\node[enc=orange] (ec) at (\encX, \yC) {text encoder\\$\phi^t$};

% inputs -> encoders (image and text continue straight; caption from its box)
\draw[iarr] (qv.east) -- (ev.west);
\draw[tarr] (qt.east) -- (et.west);
\draw[carr] (cap.east) -- (ec.west);

% =============================================================================
% CENTERING OPERATORS  (x = 8.6)
% =============================================================================
\def\ctrX{9.1}
\node[op] (muv) at (\ctrX, \yI) {$-\boldsymbol{\mu}^v$};
\node[op] (mut) at (\ctrX, \yT) {$-\boldsymbol{\mu}^t$};
\node[op] (muc) at (\ctrX, \yC) {$-\boldsymbol{\mu}^t$};

\draw[iarr] (ev.east) -- (muv.west);
\draw[tarr] (et.east) -- (mut.west);
\draw[carr] (ec.east) -- (muc.west);

% centred query labels
\node[right=0.30cm of muv, font=\large, text=blue!60!black]   (qvb) {$\bar{q}^v$};
\node[right=0.30cm of mut, font=\large, text=teal!65!black]   (qtb) {$\bar{q}^t$};
\node[right=0.30cm of muc, font=\large, text=orange!70!black] (qcb) {$\bar{c}$};

\draw[iarr] (muv.east) -- (qvb);
\draw[tarr] (mut.east) -- (qtb);
\draw[carr] (muc.east) -- (qcb);

% =============================================================================
% IMAGE DATABASE  (thumbnail stack, x = 11.6, above image branch)
% =============================================================================
\def\dbX{12.1}
\node[imgnode=gray] (dbA) at (\dbX-0.18, 5.55) {\includegraphics[width=1.15cm]{\pipeDbA}};
\node[imgnode=gray] (dbB) at (\dbX,      5.35) {\includegraphics[width=1.15cm]{\pipeDbB}};
\node[imgnode=gray] (dbC) at (\dbX+0.18, 5.15) {\includegraphics[width=1.15cm]{\pipeDbC}};
\node[right=0.12cm of dbC, align=left, font=\small] {image\\dataset $\mathcal{X}$};

% =============================================================================
% DOT PRODUCT OPERATORS  (x = dbX, aligned with database)
% =============================================================================
\node[dotop] (dv) at (\dbX, \yI) {$\cdot$};
\node[dotop] (dt) at (\dbX, \yT) {$\cdot$};
\node[dotop] (dc) at (\dbX, \yC) {$\cdot$};

\draw[iarr] (qvb.east) -- (dv.west);
\draw[tarr] (qtb.east) -- (dt.west);
\draw[carr] (qcb.east) -- (dc.west);

% database -> dot products: one vertical spine feeding all three
\draw[garr] (dbC.south) -- (\dbX, \yI+0.34);
\draw[gray!52, line width=0.80pt] (\dbX, \yI+0.34) -- (\dbX, \yC+0.34);
\draw[garr] (\dbX, \yI+0.34) -- (dv.north);
\draw[garr] (\dbX, \yT+0.34) -- (dt.north);
\draw[garr] (\dbX, \yC+0.34) -- (dc.north);
\node[right=2pt, font=\scriptsize, text=gray!60] at (\dbX, \yT+0.60) {$\bar{x}^v_i$};

% =============================================================================
% SIMILARITY SCORES  (x = 13.3)
% =============================================================================
\def\simX{13.8}
\node[font=\large, text=blue!60!black]   (sv) at (\simX, \yI) {$s^v_i$};
\node[font=\large, text=teal!65!black]   (st) at (\simX, \yT) {$s^t_i$};
\node[font=\large, text=orange!70!black] (sc) at (\simX, \yC) {$s^c_i$};

\draw[iarr] (dv.east) -- (sv);
\draw[tarr] (dt.east) -- (st);
\draw[carr] (dc.east) -- (sc);

% =============================================================================
% CLAMP BOXES + PRODUCT (soft logical AND)  (x = 14.8 .. 16.4)
% =============================================================================
\def\clX{15.3}
\node[draw=blue!45!gray, fill=blue!6, rounded corners=2pt,
      minimum width=1.45cm, minimum height=0.60cm, font=\small, align=center]
  (clv) at (\clX, \yI) {clamp};
\node[draw=teal!45!gray, fill=teal!6, rounded corners=2pt,
      minimum width=1.45cm, minimum height=0.60cm, font=\small, align=center]
  (clt) at (\clX, \yT) {clamp};
\node[draw=orange!55, fill=orange!6, rounded corners=2pt,
      minimum width=1.45cm, minimum height=0.60cm, font=\small, align=center]
  (clc) at (\clX, \yC) {clamp};

\draw[iarr] (sv.east) -- (clv.west);
\draw[tarr] (st.east) -- (clt.west);
\draw[carr] (sc.east) -- (clc.west);

\node[prodop] (prod) at (16.9, \yT) {$\times$};
\node[above=1pt of prod, lbl=gray!70!black] {soft AND};
\draw[iarr] (clv.east) -- ++(0.30,0) |- (prod.north);
\draw[tarr] (clt.east) -- (prod.west);
\draw[carr] (clc.east) -- ++(0.30,0) |- (prod.south);

% =============================================================================
% FINAL SCORE + OUTPUT  (x = 17.5 ..)
% =============================================================================
\node[font=\large, right=0.35cm of prod] (sfinal) {$\tilde{s}^f_i$};
\draw[garr, line width=0.90pt] (prod.east) -- (sfinal);

\node[imgnode=green, right=0.70cm of sfinal] (out)
  {\includegraphics[width=1.55cm]{\pipeTop}};
\node[below=1pt of out, sub] {top-1 retrieved};
\draw[garr, line width=0.90pt] (sfinal.east) -- (out.west);

% =============================================================================
% LEGEND  (bottom, below the caption branch)
% =============================================================================

  \draw[iarr, line width=1.2pt] (0,0.40) -- ++(0.75,0)
    node[right, font=\small, text=black] {image branch};
  \draw[tarr, line width=1.2pt] (3.1,0.40) -- ++(0.75,0)
    node[right, font=\small, text=black] {text branch};
  \draw[carr, line width=1.2pt] (6.1,0.40) -- ++(0.75,0)
    node[right, font=\small, text=black] {caption branch};
\end{scope}

\end{tikzpicture}
