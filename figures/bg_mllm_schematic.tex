\def\bgRef{figures/bg_mllm_samples/ref}
\def\bgQueryText{as a huge balloon}
\def\bgCaptionText{``Lightyear as a huge balloon with a green suit, purple helmet, and spaceship helmet.''}

\begin{tikzpicture}[
  >=Stealth, line cap=round, line join=round,
  font=\large,
  imgnode/.style   = {draw=#1, line width=1.6pt, inner sep=0pt},
  imgnode/.default = gray,
  enc/.style       = {draw=#1!65!black, fill=#1!10, rounded corners=3pt,
                       minimum width=2.5cm, minimum height=1.0cm,
                       align=center, inner sep=3pt, font=\large},
  enc/.default     = gray,
  mllmbox/.style   = {draw=orange!70!black, fill=orange!12, rounded corners=3pt,
                       minimum width=2.4cm, minimum height=1.4cm,
                       align=center, font=\large},
  txtph/.style     = {draw=teal!60!black, fill=white, rounded corners=3pt,
                       line width=1.1pt, text width=2.8cm, minimum height=1.3cm,
                       align=center, font=\large\itshape, inner sep=6pt},
  capph/.style     = {draw=orange!65!black, fill=orange!8, rounded corners=3pt,
                       line width=1.1pt, text width=4.6cm, align=center,
                       font=\normalsize\itshape, inner sep=6pt},
  sop/.style       = {draw=gray!45!black, fill=white, circle,
                       minimum size=1.0cm, inner sep=1pt, font=\small, align=center},
  sub/.style       = {font=\normalsize},
  footlbl/.style   = {font=\small, text=gray!45!black},
  iarr/.style      = {->, draw=blue!60!black,   line width=1.2pt},
  tarr/.style      = {->, draw=teal!65!black,   line width=1.2pt},
  carr/.style      = {->, draw=orange!75!black, line width=1.2pt},
  garr/.style      = {->, draw=gray!50,         line width=1.0pt},
]

\def\yTop{1.7}
\def\yBot{-1.7}

% ---- inputs ----
\node[imgnode=blue!60!black] (qv) at (0, \yTop) {\includegraphics[width=2.9cm]{\bgRef}};
\node[below=3pt of qv, sub] (qvlbl) {image query $q^v$};
\node[below=2pt of qvlbl, footlbl, align=center, text width=3.1cm]
  {(tiled: high-res crops + thumbnail)};

\node[txtph] (qt) at (0, \yBot) {\bgQueryText};
\node[below=3pt of qt, sub] {text query $q^t$};

% ---- vision side: ViT -> pixel shuffle -> MLP projector ----
\node[enc=blue] (vit) at (3.6, \yTop) {vision encoder\\(ViT)};
\node[sop] (pshuf) at (6.1, \yTop) {pixel\\shuffle};
\node[footlbl, below=2pt of pshuf] {4$\times$ downsample};
\node[enc=teal] (mlp) at (8.6, \yTop) {MLP\\projector};

\draw[iarr] (qv.east)   -- (vit.west);
\draw[iarr] (vit.east)  -- (pshuf.west);
\draw[garr] (pshuf.east) -- (mlp.west);

% ---- LLM: receives visual tokens (top) and text prompt (bottom) ----
\node[mllmbox] (llm) at (11.6, 0) {LLM};
\draw[iarr] (mlp.east) -- ++(0.35,0) |- (llm.north west);
\draw[tarr] (qt.east)  -- ++(0.35,0) |- (llm.south west);

% ---- output caption ----
\node[capph] (cap) at (16.0, 0) {\bgCaptionText};
\node[below=3pt of cap, sub] {target caption $c$};
\draw[carr] (llm.east) -- (cap.west);
