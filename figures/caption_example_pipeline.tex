\def\capexRef{figures/caption_example_samples/ref}
\def\capexTop{figures/caption_example_samples/top1}
\def\capexQueryText{as an iconic rooftop sign}
\def\capexCaptionText{``Cartoon character in a beige coat and\,\dots''}

\begin{tikzpicture}[
  >=Stealth, line cap=round, line join=round,
  font=\large,
  imgnode/.style   = {draw=#1, line width=1.6pt, inner sep=0pt},
  imgnode/.default = gray,
  mllmbox/.style   = {draw=orange!70!black, fill=orange!12, rounded corners=3pt,
                       minimum width=2.3cm, minimum height=1.3cm,
                       align=center, font=\large},
  txtph/.style     = {draw=teal!60!black, fill=white, rounded corners=3pt,
                       line width=1.1pt, text width=2.9cm, minimum height=1.3cm,
                       align=center, font=\large\itshape, inner sep=6pt},
  capph/.style     = {draw=orange!65!black, fill=orange!8, rounded corners=3pt,
                       line width=1.1pt, text width=3.6cm, minimum height=1.3cm,
                       align=center, font=\large\itshape, inner sep=6pt},
  sub/.style       = {font=\large},
  carr/.style      = {->, draw=orange!75!black, line width=1.2pt},
  garr/.style      = {->, draw=gray!45,         line width=1.0pt},
]

\def\yI{1.6}
\def\yT{-1.6}

\node[imgnode=blue!60!black] (qv) at (0, \yI) {\includegraphics[width=3.2cm]{\capexRef}};
\node[below=3pt of qv, sub] {image query $q^v$};

\node[txtph] (qt) at (0, \yT) {\capexQueryText};
\node[below=3pt of qt, sub] {text query $q^t$};

\node[mllmbox] (mllm) at (3.2, 0) {MLLM};
\draw[carr] (qv.south east) -- ++(0,-0.35) -- (mllm.north west);
\draw[carr] (qt.north east) -- ++(0,0.35)  -- (mllm.south west);

\node[capph] (cap) at (6.6, 0) {\capexCaptionText};
\node[below=3pt of cap, sub] {target caption $c$};
\draw[carr] (mllm.east) -- (cap.west);

\node[imgnode=green!55!black] (out) at (13.4, 0)
  {\includegraphics[width=3.2cm]{\capexTop}};
\node[below=3pt of out, sub] {top-1 retrieved};
\draw[garr, line width=1.1pt] (cap.east) -- (out.west)
  node[midway, above=7pt, font=\footnotesize, text=gray!40!black] {embed + retrieve}
  node[midway, below=7pt, font=\footnotesize, text=gray!40!black] {rank by $s^c_i$};

\end{tikzpicture}
