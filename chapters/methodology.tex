% =====================================================================

%     s_cap  = sim(phi_txt(c),  phi_img(x))
% - Define the triplet score as the clamped product (fill the equation):
% - Explain WHY: product = logical AND (a good match must satisfy all three);
%   clamp(.) avoids sign flips / negative-times-negative artefacts.
% - Contrast with the duplet baseline (Section~\ref{subsect:method_duplet}).
\begin{equation}\label{eq:triplet}
    s_{\text{triplet}}(q, x) =
    \clamp\!\big(s_{\text{cap}}\big)\cdot
    \clamp\!\big(s_{\text{txt}}\big)\cdot
    \clamp\!\big(s_{\text{img}}\big).
\end{equation}
